	public void findingUniqueString(String str)
	{
		
		try{
			
		for(int i=0;i<str.length();i++)
		{
			int count=1;
			for(int j=i+1;j<str.length();j++)
			{
				//System.out.println("  " +count);
			if(str.charAt(i)==str.charAt(j))
			{
				str.charAt(j);
				result=++count;
				
			}
			
			}
			
			
		}
		System.out.println(result);
		
		}
		catch(Exception e)
		{
			
		}

	}
